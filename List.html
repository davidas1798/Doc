def [](i)
	if i.is_a?Integer
		if i == 0
			return @head.value
		elsif i == (@size - 1)
			return @tail.value
		elsif i < @size - 1 && i > 0
			puntero = @head
			i.times{puntero = puntero.next}
			return puntero.value
		else 
			return nil
		end
	end
end

def insert(item)
	item = Node.new(item, nil, nil)
	if(empty)
		@head = item
		item.prev = nil
	else
		@tail.next = item
		item.prev = @tail
	end

	@tail = item
	item.next = nil
	@size += 1
end

def to_s

	cadena =""
